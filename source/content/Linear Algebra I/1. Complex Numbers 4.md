$$
\begin{gather*}
az^2 + bz + c = 0 \\
z = \frac{b \pm \sqrt {b^2 - 4ac}}{2a} \\ \\

D = b^2 - 4ac \\
D = re^{i\theta} = re^{i(\theta + 2n\pi)} \\
\sqrt{D} = \sqrt{r}e^{i(\frac{\theta}{2}+n\pi)} \\
\text{if } D \in \mathbb{R}, \\ \\
\sqrt{D} = \pm i\sqrt{-D}
\end{gather*}
$$

$$
\begin{align}
z^3 &= -i \\
&= e^{-\frac{i\pi}{2}} \\
&= e^{-i(\frac{\pi}{2} + 2n\pi)} \\ \\ 

z &= e^{-i(\frac{\pi}{6} + \frac{2n\pi}{3})} \\
\end{align}
$$
